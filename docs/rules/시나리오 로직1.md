  Supabase 테이블 설계

  CREATE TABLE scenarios (
    id            UUID    DEFAULT gen_random_uuid() PRIMARY KEY,
    template_id   TEXT    NOT NULL,
    style         TEXT    NOT NULL,
    topic         TEXT    NOT NULL,
    script        TEXT    NOT NULL,
    scene_count   INT,
    estimated_sec FLOAT,
    viral_seed    FLOAT,          -- 씨드가 된 크롤 데이터 점수
    used          BOOLEAN DEFAULT FALSE,
    used_at       TIMESTAMPTZ,
    created_at    TIMESTAMPTZ DEFAULT NOW()
  );

  -- 성능 핵심 인덱스 (재고 조회 속도)
  CREATE INDEX idx_stock
    ON scenarios(template_id, style, used)
    WHERE used = FALSE;
