## 5. URL 인풋: 메타태그 파싱 + 룰

## 5.1 URL 처리 플로우

1. 사용자가 URL 공유/입력
    
2. 백그라운드에서 HTTP GET
    
3. HTML 파싱 후, 아래 우선순위로 제목 결정
    
    - `og:title`
        
    - `<title>`
        
    - 없으면 `도메인 + 날짜` 패턴
        

## 5.2 의존성 예시

text

`implementation "com.squareup.okhttp3:okhttp:4.12.0" implementation "org.jsoup:jsoup:1.17.2"`

## 5.3 파싱 코드 예시

kotlin

`suspend fun fetchTitleFromUrl(url: String): String? = withContext(Dispatchers.IO) {     try {        val response = OkHttpClient().newCall(            Request.Builder().url(url).build()        ).execute()         val body = response.body?.string() ?: return@withContext null        val doc = Jsoup.parse(body)         // 1) og:title        doc.selectFirst("meta[property=og:title]")?.attr("content")?.takeIf { it.isNotBlank() }            ?: // 2) <title>            doc.title().takeIf { it.isNotBlank() }    } catch (e: Exception) {        null    } }`

## 5.4 최종 제목 결정

kotlin

`suspend fun generateTitleForUrl(url: String, createdAt: Long): String {     val fetched = fetchTitleFromUrl(url)    if (!fetched.isNullOrBlank()) return fetched.trim()     // 메타태그 실패 시    val host = try {        URI(url).host?.removePrefix("www.") ?: url    } catch (e: Exception) {        url    }     return "${host} · ${createdAt.toDateLabel()}" }`

## 6. 텍스트 인풋: 앞부분 잘라서 제목 생성

## 6.1 규칙

- 우선순위:
    
    1. 첫 줄 (줄바꿈 전까지)
        
    2. 없으면 전체에서 앞 25~30자
        
    3. 모두 짧거나 공백이면 “메모 · 날짜”
        

## 6.2 코드 예시

kotlin

`fun generateTitleForText(rawText: String, createdAt: Long): String {     val trimmed = rawText.trim()    if (trimmed.isEmpty()) {        return TitleGenerator.defaultTitle("메모", createdAt)    }     val firstLine = trimmed.lineSequence().firstOrNull()?.trim().orEmpty()    val base = if (firstLine.isNotEmpty()) firstLine else trimmed     val maxLen = 30    val title = if (base.length > maxLen) {        base.substring(0, maxLen) + "…"    } else {        base    }     return title }`

## 7. 이미지 인풋: 캡션/카테고리 기반 제목

## 10. 특징 요약 (문서용 정리)

- **서버/AI 없음**: 모든 처리는 로컬에서, 룰 + 간단 HTML 파싱만 사용.
    
- **자동 제목 규칙**:
    
    - URL: `og:title` → `<title>` → 실패 시 `도메인 + 날짜`.
        
    - 텍스트: 첫 줄/앞 30자를 잘라 제목, 없으면 “메모 · 날짜”.
        
    - 이미지: 캡션 있으면 캡션, 없으면 “카테고리명 · 날짜” 또는 “캡처 · 날짜”.
        
- **생성일**: Note 생성 시 epoch millis 저장, UI에서 `yyyy.MM.dd` 포맷으로 사용.

## 4-3. URL 제목 생성

`util/UrlTitleGenerator.kt`:

kotlin
```
package com.yourname.linkdrawer.util

import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.withContext
import okhttp3.OkHttpClient
import okhttp3.Request
import org.jsoup.Jsoup
import java.net.URI

object UrlTitleGenerator {

    private val client = OkHttpClient()

    suspend fun generateTitleForUrl(url: String, createdAt: Long): String =
        withContext(Dispatchers.IO) {
            val fetched = fetchTitleFromUrl(url)
            if (!fetched.isNullOrBlank()) return@withContext fetched.trim()

            val host = try {
                URI(url).host?.removePrefix("www.") ?: url
            } catch (e: Exception) {
                url
            }
            "${host} · ${createdAt.toDateLabel()}"
        }

    private fun fetchTitleFromUrl(url: String): String? {
        return try {
            val response = client.newCall(
                Request.Builder().url(url).build()
            ).execute()

            val body = response.body?.string() ?: return null
            val doc = Jsoup.parse(body)

            doc.selectFirst("meta[property=og:title]")?.attr("content")?.takeIf { it.isNotBlank() }
                ?: doc.title().takeIf { it.isNotBlank() }
        } catch (e: Exception) {
            null
        }
    }
}

```

```
package com.yourname.linkdrawer.util

object TextTitleGenerator {

    fun generateTitleForText(rawText: String, createdAt: Long): String {
        val trimmed = rawText.trim()
        if (trimmed.isEmpty()) {
            return TitleGenerator.defaultTitle("메모", createdAt)

        } else {
            base
        }

        return title
    }
}
```

## 5. Note 저장 로직 뼈대 (ViewModel 관점)

실제 UI는 나중에 천천히 만들고, 우선 “저장 로직”만 만들자.

`domain/NoteRepository.kt` (새 패키지 `domain`):

kotlin
```
package com.yourname.linkdrawer.domain

import com.yourname.linkdrawer.data.db.AppDatabase
import com.yourname.linkdrawer.data.entity.Note
import com.yourname.linkdrawer.data.entity.NoteType
import com.yourname.linkdrawer.util.ImageTitleGenerator
import com.yourname.linkdrawer.util.TextTitleGenerator
import com.yourname.linkdrawer.util.UrlTitleGenerator
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.withContext

class NoteRepository(private val db: AppDatabase) {

    private val noteDao = db.noteDao()
    private val categoryDao = db.categoryDao()

    suspend fun saveLink(url: String, categoryId: Long?) {
        val now = System.currentTimeMillis()
        val title = UrlTitleGenerator.generateTitleForUrl(url, now)

        val note = Note(
            type = NoteType.LINK,
            title = title,
            content = null,
            url = url,
            imagePath = null,
            categoryId = categoryId,
            createdAt = now,
            updatedAt = now
        )
        withContext(Dispatchers.IO) {
            noteDao.insert(note)
        }
    }

    suspend fun saveText(text: String, categoryId: Long?) {
        val now = System.currentTimeMillis()
        val title = TextTitleGenerator.generateTitleForText(text, now)

        val note = Note(
            type = NoteType.TEXT,
            title = title,
            content = text,
            url = null,
            imagePath = null,
            categoryId = categoryId,
            createdAt = now,
            updatedAt = now
        )
        withContext(Dispatchers.IO) {
            noteDao.insert(note)
        }
    }

    suspend fun saveImage(
        localImagePath: String,
        userCaption: String?,
        categoryId: Long?,
        categoryName: String?
    ) {
        val now = System.currentTimeMillis()
        val title = ImageTitleGenerator.generateTitleForImage(userCaption, categoryName, now)

        val note = Note(
            type = NoteType.IMAGE,
            title = title,
            content = userCaption,
            url = null,
            imagePath = localImagePath,
            categoryId = categoryId,
            createdAt = now,
            updatedAt = now
        )
        withContext(Dispatchers.IO) {
            noteDao.insert(note)
        }
    }
}
```

## 6-2. MainActivity 코드

kotlin
```
package com.yourname.linkdrawer

import androidx.appcompat.app.AppCompatActivity
import android.os.Bundle
import android.widget.Button
import android.widget.EditText
import android.widget.Toast
import androidx.lifecycle.lifecycleScope
import com.yourname.linkdrawer.data.db.AppDatabase
import com.yourname.linkdrawer.domain.NoteRepository
import kotlinx.coroutines.launch

class MainActivity : AppCompatActivity() {

    private lateinit var repository: NoteRepository

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        val db = AppDatabase.getInstance(this)
        repository = NoteRepository(db)

        val etUrl = findViewById<EditText>(R.id.etUrl)
        val btnSave = findViewById<Button>(R.id.btnSaveUrl)

        btnSave.setOnClickListener {
            val url = etUrl.text.toString()
            if (url.isBlank()) {
                Toast.makeText(this, "URL을 입력하세요", Toast.LENGTH_SHORT).show()
                return@setOnClickListener
            }

            lifecycleScope.launch {
                repository.saveLink(url, categoryId = null)
                runOnUiThread {
                    Toast.makeText(this@MainActivity, "저장 완료", Toast.LENGTH_SHORT).show()
                }
            }
        }
    }
}
```

**왕초보 기준 “Android Studio 사용 + 로컬 저장 + 자동 제목 생성” MVP 골격 문서**

방법 1: SDK Manager에서 재설치 (가장 쉬움)
팝업창에서 하지 말고, 안드로이드 스튜디오 설정 메뉴에서 직접 설치해 봅니다.
1.
안드로이드 스튜디오 상단 메뉴에서 Tools > SDK Manager를 클릭합니다.
2.
나오는 창 중간에 있는 탭 중 SDK Tools를 선택합니다.
3.
목록에서 Android Emulator Hypervisor Driver (installer) 를 찾습니다.
◦
체크가 되어 있다면: 체크를 해제하고 하단의 Apply를 눌러 삭제합니다. 그 후 다시 체크하고 Apply를 눌러 재설치합니다.
◦
체크가 안 되어 있다면: 체크하고 Apply를 눌러 설치합니다.
4.
설치가 완료되면 다시 실행(Run ▶) 해보세요.
방법 2: 컴퓨터의 가상화(Virtualization) 기능 켜기 (가장 흔한 원인)
이 드라이버는 컴퓨터 CPU의 **가상화 기술(VT-x 또는 SVM)**이 켜져 있어야 설치됩니다. 만약 방법 1로도 안 된다면, 컴퓨터 바이오스(BIOS) 설정에서 이 기능이 꺼져 있을 확률이 90%입니다.
•
컴퓨터를 재부팅하면서 F2나 Del 키를 눌러 바이오스 진입 후, Virtualization Technology (또는 SVM Mode) 항목을 찾아 Enable(켜짐) 로 변경해야 합니다. (컴퓨터 모델마다 다르니 유튜브에 '바이오스 가상화 켜기'를 검색해 보시면 쉽습니다.)
방법 3: 실제 폰 연결하기 (추천)
에뮬레이터 설정이 너무 복잡하고 어렵다면, 가지고 계신 안드로이드 폰을 컴퓨터에 USB로 연결해서 테스트하는 것이 훨씬 빠르고 정확합니다.
1.
폰의 설정 > 휴대전화 정보 > 빌드 번호를 7번 연속 터치하여 '개발자 모드'를 켭니다.
2.
설정 > 개발자 옵션에 들어가서 USB 디버깅을 켭니다.
3.
USB 케이블로 컴퓨터에 연결하면 안드로이드 스튜디오 상단에 폰 이름이 뜹니다. 그걸 선택하고 실행(▶) 하시면 됩니다.
